音频 DSP

在 Cortex-M 上实现音频信号处理。使用 CMSIS-DSP 库进行 EQ、压缩、滤波等操作。理解定点/浮点优化。

5.1 CMSIS-DSP — ARM 官方 DSP 库

CMSIS-DSP 是 ARM 提供的免费 DSP 库，针对 Cortex-M 的 FPU 和 SIMD 指令做了汇编级优化。这是嵌入式音频处理的标准工具。

5.2 Biquad 滤波器 — 音频 EQ 的基础

Biquad（双二阶）滤波器是音频 EQ 的基本单元。每个频段的 EQ 就是一个 Biquad 滤波器。多个级联 = 多段 EQ。

audio_eq.c — 3 段参数 EQ 实现 
#include "arm_math.h"

// Biquad 滤波器状态和实例
#define NUM_EQ_BANDS 3  // 低频、中频、高频

static float32_t eq_state[NUM_EQ_BANDS * 4];  // 每个 biquad 需要 4 个状态变量
static arm_biquad_casd_df1_inst_f32 eq_instance;

// ① 计算 Biquad 系数（根据滤波器类型和参数）
void calc_biquad_coeff(float *coeffs, float freq, float Q, float gain_db, int type) {
    float A = powf(10.0f, gain_db / 40.0f);  // 增益转换为线性
    float w0 = 2.0f * PI * freq / SAMPLE_RATE;
    float alpha = sinf(w0) / (2.0f * Q);
    float cosw0 = cosf(w0);

    switch (type) {
        case EQ_PEAKING:  // 峰值 EQ（最常用）
            coeffs[0] = 1.0f + alpha * A;       // b0
            coeffs[1] = -2.0f * cosw0;           // b1
            coeffs[2] = 1.0f - alpha * A;        // b2
            coeffs[3] = -2.0f * cosw0;           // a1 (注意符号)
            coeffs[4] = 1.0f + alpha / A;        // a2
            break;
        case EQ_LOWSHELF:  // 低频搁架
            // ... 类似公式，不同的系数计算
            break;
    }
    // 归一化：所有系数除以 a0
    float a0 = 1.0f + alpha / A;
    for (int i = 0; i < 5; i++) coeffs[i] /= a0;
}

// ② 初始化 3 段 EQ
void eq_init(void) {
    float32_t coeffs[NUM_EQ_BANDS * 5];  // 每个 biquad 5 个系数

    // 低频段：200Hz, Q=1.0, +3dB
    calc_biquad_coeff(&coeffs[0], 200.0f, 1.0f, 3.0f, EQ_PEAKING);
    // 中频段：1000Hz, Q=1.5, -2dB
    calc_biquad_coeff(&coeffs[5], 1000.0f, 1.5f, -2.0f, EQ_PEAKING);
    // 高频段：5000Hz, Q=1.0, +4dB
    calc_biquad_coeff(&coeffs[10], 5000.0f, 1.0f, 4.0f, EQ_PEAKING);

    arm_biquad_cascade_df1_init_f32(&eq_instance,
        NUM_EQ_BANDS, coeffs, eq_state);
}

// ③ 在音频回调中使用 EQ
void eq_process(float32_t *in, float32_t *out, uint32_t block_size) {
    // CMSIS-DSP 优化过的 Biquad 级联滤波
    // 利用 FPU 的 SIMD 指令，比手写 C 快 3-5 倍
    arm_biquad_cascade_df1_f32(&eq_instance, in, out, block_size);
}

// ④ 完整的音频处理链
void audio_callback(int16_t *in, int16_t *out, uint32_t size) {
    float32_t float_in[256], float_out[256];

    // Q15 → float
    arm_q15_to_float(in, float_in, size);

    // EQ 处理
    eq_process(float_in, float_out, size);

    // float → Q15
    arm_float_to_q15(float_out, out, size);
}
① Biquad 系数计算是 EQ 的核心数学。Audio EQ Cookbook（Robert Bristow-Johnson）是经典参考。不同类型（峰值、低架、高架、带通）有不同公式。
② 级联（Cascade）：3 段 EQ = 3 个 Biquad 串联。CMSIS-DSP 的 cascade 函数一次性处理所有级，比单独调用 3 次更高效。
③ FPU 加速：arm_biquad_cascade_df1_f32 在 M4F 上用硬件 FPU，每秒可处理数百万采样。256 samples 的 Biquad 处理只需 ~20μs。
④ 数据格式转换：Codec 输出 int16（Q15），DSP 用 float32。这是标准做法 — float 运算在现代 Cortex-M 上比定点更快（有 FPU）。
不可替代

EQ 类型 + Q 值的听觉判断是纯领域知识

Peaking、Low-shelf、High-shelf、Bandpass——这些不是代码概念，是音频工程数十年的积累。Q=0.707 是 Butterworth 平坦响应，Q>1 会产生谐振峰。AI 能生成系数计算代码，但如果你分不清滤波器类型，EQ 不会对。

5.3 动态范围压缩（DRC）

DRC 用于自动控制音量 — 声音太大时压低，太小时提升。助听器、TWS 耳机、智能眼镜都用得到。

drc.c — 简单的动态范围压缩器 
typedef struct {
    float threshold;    // 阈值 (dB) — 超过这个开始压缩
    float ratio;        // 压缩比 — 4:1 表示超过阈值的部分缩小到 1/4
    float attack;       // 启动时间 (s) — 检测到超阈值后多快开始压缩
    float release;      // 释放时间 (s) — 信号回到阈值下后多快停止压缩
    float envelope;     // 当前包络值（内部状态）
} compressor_t;

float compressor_process(compressor_t *c, float input) {
    // ① 计算输入信号电平（dB）
    float input_level = 20.0f * log10f(fabsf(input) + 1e-10f);

    // ② 计算目标增益
    float gain_db = 0.0f;
    if (input_level > c->threshold) {
        // 超过阈值的部分按比例压缩
        float excess = input_level - c->threshold;
        gain_db = -excess * (1.0f - 1.0f / c->ratio);
    }

    // ③ 平滑增益变化（attack/release 包络跟随）
    float coeff;
    if (gain_db < c->envelope) {
        // 信号在增大 → 使用 attack 系数（快速响应）
        coeff = expf(-1.0f / (c->attack * SAMPLE_RATE));
    } else {
        // 信号在减小 → 使用 release 系数（慢速释放）
        coeff = expf(-1.0f / (c->release * SAMPLE_RATE));
    }
    c->envelope = gain_db + coeff * (c->envelope - gain_db);

    // ④ 应用增益
    float gain_linear = powf(10.0f, c->envelope / 20.0f);
    return input * gain_linear;
}

// 典型配置：语音处理
compressor_t voice_compressor = {
    .threshold = -20.0f,   // -20dB 以下不压缩
    .ratio = 4.0f,         // 4:1 压缩比
    .attack = 0.005f,      // 5ms 启动
    .release = 0.050f,     // 50ms 释放
    .envelope = 0.0f,
};
①② 增益计算：输入在阈值以下不压缩；超过阈值的部分除以压缩比。4:1 意味着超过 20dB 的信号被压到 5dB。
③ Attack/Release 是关键。Attack 太快会失真（"泵浦"效应），太慢会漏掉瞬态。5ms attack + 50ms release 是语音处理的经典参数。
不可替代

Attack/Release 来自心理声学

5ms attack + 50ms release 是语音处理的经典参数——但这不是代码知识，是几十年的听觉工程积累。Attack 太快 = 泵浦效应失真，Release 太快 = 呼吸感。AI 能生成压缩器代码，但无法替你判断"这个声音压得对不对"。

5.4 主动降噪（ANC）基础

ANC 是智能音频设备的核心技术。原理：用参考麦克风采集环境噪声，生成反相信号叠加到输出，实现降噪。Feedforward ANC 是最简单的入门方案。

anc_lms.c — NLMS 自适应滤波器（简化版） 
#define FILTER_LEN 64  // 自适应滤波器长度

typedef struct {
    float w[FILTER_LEN];      // 滤波器系数（权重）
    float x_buf[FILTER_LEN];  // 参考信号缓冲（环形）
    int idx;                   // 环形缓冲区索引
    float mu;                  // 步长（收敛速度 vs 稳定性）
} nlms_filter_t;

// NLMS (Normalized LMS) — 最常用的自适应算法
float nlms_process(nlms_filter_t *f, float ref, float error) {
    // ① 存储新的参考信号
    f->x_buf[f->idx] = ref;

    // ② 计算滤波器输出 y(n) = W^T · X
    float y = 0.0f;
    for (int i = 0; i < FILTER_LEN; i++) {
        int j = (f->idx - i + FILTER_LEN) % FILTER_LEN;
        y += f->w[i] * f->x_buf[j];
    }

    // ③ 计算参考信号功率（归一化用）
    float power = 1e-10f;  // 防止除零
    for (int i = 0; i < FILTER_LEN; i++) {
        power += f->x_buf[i] * f->x_buf[i];
    }

    // ④ 更新滤波器系数
    // NLMS: W(n+1) = W(n) + (μ / power) * e(n) * X(n)
    float step = f->mu / power;
    for (int i = 0; i < FILTER_LEN; i++) {
        int j = (f->idx - i + FILTER_LEN) % FILTER_LEN;
        f->w[i] += step * error * f->x_buf[j];
    }

    // 更新索引
    f->idx = (f->idx + 1) % FILTER_LEN;

    return y;  // 返回反相噪声信号
}
① 参考信号 = 环境噪声（由外侧麦克风采集）。误差信号 = 用户耳道内实际听到的声音（内侧麦克风采集）。
②③ 归一化 (Normalized)：LMS 的改进版。除以信号功率使得步长自适应 — 信号大时步长小，信号小时步长大。收敛更稳定。
④ 系数更新是自适应滤波的核心。每次采样都更新所有 64 个系数，计算量 = 2 × 64 × 64 = 8192 MAC/采样。在 48kHz 下需要 394 MIPS — M7 可以做到，M4F 勉强。

ANC 的挑战：① 延迟必须极低（< 2ms），否则反相信号和实际噪声不同步；② 声学路径变化（耳机位置移动）要求滤波器实时适应；③ 反馈稳定性（Feedforward 没有这个问题，Feedback ANC 有）。

不可替代

延迟约束 + MIPS 预算是可行性分析

M4F 做 64-tap NLMS@48kHz 需要 ~394 MIPS，但 M4F 只有 ~250 DMIPS。这个算术决定"做不了"，不是代码问题。ANC 的延迟必须 <2ms，这个硬约束决定了 buffer size、tap 数、MCU 选型。AI 无法做这种系统级可行性判断。

5.5 麦克风阵列与波束成形

智能眼镜通常有 2-4 个麦克风做波束成形 — 增强 speaking 方向的声音，抑制其他方向。核心是延时叠加（Delay-and-Sum）。

beamform.c — 2 麦延时叠加波束成形 
#define NUM_MICS 2
#define MAX_DELAY_SAMPLES 48  // 最大延时 (采样数)，对应 ~34cm 距离

// 延时叠加波束成形
void beamform_delay_sum(float *mic_in[NUM_MICS], float *output,
                        int *delays, uint32_t block_size) {
    for (uint32_t n = 0; n < block_size; n++) {
        float sum = 0.0f;
        for (int m = 0; m < NUM_MICS; m++) {
            // 对每个麦克风的信号施加延时
            int delayed_n = n - delays[m];
            if (delayed_n >= 0) {
                sum += mic_in[m][delayed_n];
            }
        }
        output[n] = sum / NUM_MICS;  // 平均
    }
}

// 计算目标方向的延时
void calc_steering_delays(float angle_deg, int *delays,
                          float mic_spacing_m) {
    // 声速 343 m/s
    // 延时 = spacing × cos(angle) / speed_of_sound
    float speed = 343.0f;
    for (int m = 0; m < NUM_MICS; m++) {
        float delay_sec = mic_spacing_m * cosf(angle_deg * PI / 180.0f) * m / speed;
        delays[m] = (int)(delay_sec * SAMPLE_RATE + 0.5f);
    }
}
